> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Python-first offline pipeline that extracts historical stat-market data from old read-only sources, normalizes it into a new local Parquet/DuckDB structure, engineers leakage-safe features, and runs walk-forward evaluation for `totalShots`, `shotsOnGoal`, and `cornerKicks`.

**Architecture:** The pipeline is split into deterministic layers: source extraction, audit, normalization, features, and backtest/reporting. Old MongoDB collections and local raw files are treated as immutable inputs; the new repo produces disposable local derived artifacts with explicit lineage and filter reasons.

**Tech Stack:** Python 3.13, `pymongo`, `duckdb`, `pandas`, `pyarrow`, `scikit-learn`, Parquet, pytest.

def flatten_unibet_backtest_docs(docs: list[dict]) -> list[dict]:
    rows: list[dict] = []
    for doc in docs:
        for line in doc.get("lines") or []:
            rows.append({
                "match_id": doc.get("matchId"),
                "league_name": doc.get("league"),
                "bet_key": line.get("betKey"),
                "stat_key": line.get("statKey"),
                "period": line.get("period"),
                "scope": line.get("scope"),
                "direction": "under" if str(line.get("condition")).lower() == "under" else "over",
                "line_value": line.get("line"),
                "odds_decimal": line.get("odds"),
                "actual_value": line.get("actual"),
                "settlement_result": "win" if line.get("win") is True else "loss" if line.get("win") is False else None,
            })
    return rows
```

def extract_stat_rows(match: dict) -> list[dict]:
    rows: list[dict] = []
    for period_block in (((match.get("matchDetails") or {}).get("statistics")) or []):
        period = period_block.get("period")
        for group in period_block.get("groups") or []:
            for item in group.get("statisticsItems") or []:
                home_value = item.get("homeValue")
                away_value = item.get("awayValue")
                total_value = None
                if isinstance(home_value, (int, float)) and isinstance(away_value, (int, float)):
                    total_value = home_value + away_value
                rows.append({
                    "match_id": match.get("matchId"),
                    "kickoff_ts": match.get("timestamp"),
                    "period": period,
                    "stat_group": group.get("groupName"),
                    "stat_item_key": item.get("key"),
                    "stat_item_name": item.get("name"),
                    "home_value": home_value,
                    "away_value": away_value,
                    "total_value": total_value,
                })
    return rows
```

from collections import Counter


def summarize_filter_reasons(rows: list[dict]) -> dict:
    filtered = Counter()
    kept = 0
    for row in rows:
        reason = row.get("filter_reason")
        if reason:
            filtered[reason] += 1
        else:
            kept += 1
    return {"kept": kept, "filtered": dict(filtered)}
```

def last_n_average(rows: list[dict], cutoff_ts: int, n: int, value_key: str) -> float | None:
    eligible = [row[value_key] for row in rows if row.get("kickoff_ts") is not None and row["kickoff_ts"] < cutoff_ts]
    if not eligible:
        return None
    selected = eligible[-n:]
    return sum(selected) / len(selected)
```
